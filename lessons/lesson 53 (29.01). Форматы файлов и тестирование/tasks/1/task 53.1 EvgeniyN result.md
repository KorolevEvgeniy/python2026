import csv

def calculate_averages(filename):
    subject_sums = {}
    subject_counts = {}
    
    with open(filename, 'r', encoding='utf-8') as file:
        reader = csv.DictReader(file)  # Используем DictReader для удобства
        for row in reader:
            for subject in reader.fieldnames[1:]:  # Пропускаем поле 'Name'
                grade = float(row[subject])
                subject_sums[subject] = subject_sums.get(subject, 0) + grade
                subject_counts[subject] = subject_counts.get(subject, 0) + 1
    
    averages = {}
    for subject in subject_sums:
        averages[subject] = subject_sums[subject] / subject_counts[subject]
    
    return averages

if __name__ == "__main__":
    averages = calculate_averages('students.csv')
    for subject, avg in averages.items():
        print(f"{subject}: {avg}")
