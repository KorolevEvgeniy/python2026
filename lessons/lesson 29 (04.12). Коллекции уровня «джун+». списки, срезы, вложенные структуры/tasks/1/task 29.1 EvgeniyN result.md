```python
# Исходные данные
base_set = ["кисть круглая", "кисть плоская", "гуашь"]
extras = ["акварель", "палитра"]

# Добавляем экстренные запасы
base_set.extend(extras)

# Выводим количество предметов
print(f"Всего позиций: {len(base_set)}")

# Выводим второй и последний элемент
print(f"Второй предмет: {base_set[1]}")
print(f"Последний предмет: {base_set[-1]}")

# Выводим список с порядковыми номерами
for idx, item in enumerate(base_set, 1):
    print(f"{idx}. {item}")
```
